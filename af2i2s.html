문제

증가연산자(++)와 감소연산자(--)의 전위연산자와 후위연산자를 실습하고 그 결과를 출력하는 프로그램입니다 실행순서를 클릭하세요 
a = 3일 때 c = ++a 적용 후 결과 a = 4, c = 4
b = 3일 때 d = b++ 적용 후 결과 b = 4, d = 3
a = 4일 때 c = --a 적용 후 결과 a = 3, c = 3
b = 4일 때 d = b-- 적용 후 결과 b = 3, d = 4

프로그램 코드

	#include <stdio.h>


1	int main()

	{

2		int a = 3, b = 3, c = 0, d = 0;


3		c = 

4		    ++a;									

5		d =  

6		    b++;									


7		printf( "a = 3일 때 c = ++a 적용 후 결과 a = %d, c = %d\n", a, c );		

8		printf( "b = 3일 때 d = b++ 적용 후 결과 b = %d, d = %d\n", b, d );		


9		c = 

10		    --a;									

11		d = 

12		    b--;									


13		printf( "a = 4일 때 c = --a 적용 후 결과 a = %d, c = %d\n", a, c );		

14		printf( "b = 4일 때 d = b-- 적용 후 결과 b = %d, d = %d\n", b, d );		


15		return 0;

	}
