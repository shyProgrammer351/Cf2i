관련신문기사 - http://news1.kr/articles/?2246990

승리팀 : 롯데
패배팀 : KT
승리점수 : 10
패배점수 : 1
승리팀 선발투수 : 김승회

맹활약 시 뽑아낸 점수 : 2
공략못한 선수 :박진형
KT는 15일 열린 프로야구 롯데와의 홈 경기에서 1 - 10점으로 크게 패하며 
홈 팬들을 실망시켰습니다. KT는 정대현을 선발로 등판시켰고 롯데는 김승회가 나섰습니다. 
롯데는 정훈이 맹활약을 펼쳤습니다. 정훈은 6회초1아웃에 맞이한 타석에서 2점을 뽑아내며, 
롯데의 9점차 승리를 이끈 일등공신이 됐습니다. KT는 박진형을 끝까지 공략하지 못하며 
안방에서 롯데에 9점차 승리를 내주었습니다.

알고리즘

프로그램시작 
   주요 경기 내용을 입력 
   입력된 주요 경기 내용을 바탕으로 스포츠 신문기사 자동 생성 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/SportsNews.c 

	#include <stdio.h>							 

	 

	// 프로그램시작 

1	int main() { 

		// 경기 내용을 저장하는 변수 초기화 

		char date[16] = ""; 

		char winningTeam[16] = ""; 

		char losingTeam[16] = ""; 

		int  winningPoint = 0; 

		int  losingPoint = 0; 

		char losingPitcher[16] = ""; 

		char winningPitcher[16] = ""; 

		char bestPlayer[16] = ""; 

		char bestPoint[16] = ""; 

		int  bestPointScore = 0; 

2		char missedPlayer[16] = ""; 

		 

		// 주요 경기 내용을 입력 

		// 경기일자 입력 

3		printf( "경기일자 : " );					 

		// 15일을 입력한다고 가정 

4		scanf( "%s", date );						 

	 

		// 경기 팀 정보 입력 

5		printf( "\n승리팀 : " );					 

		// 롯데를 입력한다고 가정 

6		scanf( "%s", winningTeam );					 

7		printf( "패배팀 : " );						 

		// KT를 입력한다고 가정 

8		scanf( "%s", losingTeam );					 

	 

		// 점수 정보 입력 

9		printf( "\n승리점수 : " );					 

		// 10점을 입력한다고 가정		 

10		scanf( "%d", &winningPoint );					 

11		printf( "패배점수 : " );					 

		// 1점을 입력한다고 가정 

12		scanf( "%d", &losingPoint ); 					 

	 

		// 선발투수 정보 입력 

13		printf( "\n승리팀 선발투수 : " );				 

		// 김승회를 입력한다고 가정 

14		scanf( "%s", winningPitcher );			 

15		printf( "패배팀 선발투수 : " );					 

		// 정대현을 입력한다고 가정 

16		scanf( "%s", losingPitcher );			 

	 

		// 기타 정보 입력 

17		printf( "\n승리팀 맹활약 선수 : " );				 

		// 정훈을 입력한다고 가정 

18		scanf( "%s", bestPlayer );					 

19		printf( "맹활약 시점 : " );					 

		// 6회초1아웃을 입력한다고 가정 

20		scanf( "%s", bestPoint );					 

21		printf( "맹활약 시 뽑아낸 점수 : " );				 

		// 2점을 입력한다고 가정 

22		scanf( "%d", &bestPointScore );				 

23		printf( "공략못한 선수 : " );					 

		// 박진형을 입력한다고 가정 

24		scanf( "%s", missedPlayer );					 

		 

		// 입력된 주요 경기 내용을 바탕으로 스포츠 신문기사 자동 생성 

25		printf( "\n\n%s는 %s일 열린 프로야구 %s와의 홈 경기에서 %d - %d점으로 크게 패하며 ",	 

		  losingTeam, date, winningTeam, losingPoint, winningPoint ); 

26		printf( "홈 팬들을 실망시켰습니다. %s는 %s을 선발로 등판시켰고 %s는 %s가 나섰습니다. ", 	 

		  losingTeam, losingPitcher, winningTeam, winningPitcher ); 

27		printf( "%s는 %s이 맹활약을 펼쳤습니다. %s은 %s에 맞이한 타석에서 %d점을 뽑아내며, ", 	 

		  winningTeam, bestPlayer, bestPlayer, bestPoint, bestPointScore ); 

28		printf( "%s의 %d점차 승리를 이끈 일등공신이 됐습니다. %s는 %s을 끝까지 공략하지 못하며 ", 	 

		  winningTeam, ( winningPoint - losingPoint ), losingTeam, missedPlayer ); 

29		printf( "안방에서 %s에 %d점차 승리를 내주었습니다.", 					 

		  winningTeam, ( winningPoint - losingPoint ) ); 

	 

		// 프로그램 종료 

30		return 0; 

	}
