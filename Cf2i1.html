알고리즘

프로그램 시작 
   친구와 대화하기 
      이름 물어보고 답하기 
      직업 물어보고 답하기 
      취미 물어보고 답하기 
      좋아하는 색 물어보고 답하기 
      친하게 지내자고 인사하기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/Chat.c

	#include <stdio.h>

		 

	// 프로그램 시작 

1	int main() {

		// 친구와 대화하기 

		// 이름 물어보고 답하기 

2		printf( "안녕? 넌 이름이 뭐야: " );

 

		char name[32] = "";

3		scanf( "%s", name );

4		printf( "만나서 반가워, %s. \n", name );

	 

		// 직업 물어보고 답하기 

5		printf( "넌 주로 뭐하고 살아: " );

		char job[32] = "";

6		scanf( "%s", job );

7		printf( "%s? 나도 마찬가지야 ㅎㅎ \n", job );

	 

		// 취미 물어보고 답하기 

8		printf( "그럼 심심할 때 뭐하니: " );

		char hobby[32] = "";

9		scanf( "%s", hobby );

10		printf( "%s? 나랑 똑같네. \n", hobby );

	 

		// 좋아하는 색 물어보고 답하기 

11		printf( "혹시 좋아하는 색은 뭐야 : " );

		char color[32] = "";

12		scanf( "%s", color );

13		printf( "%s이라구? 와! 이것도 나랑 똑같네. \n", color );

	 

		// 친하게 지내자고 인사하기 

14		printf( "앞으로 우리 친하게 지내자.\n" );

	 

		// 프로그램 종료 

15		return 0;

	}
