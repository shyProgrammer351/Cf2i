알고리즘

프로그램시작 
   친구와 대화하기 
      이름 물어보고 답하기 
      직업 물어보고 답하기 
      취미 물어보고 답하기 
      좋아하는 색 물어보고 답하기 
      친하게 지내자고 인사하기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/Chat.c

	#include <stdio.h>

	 

	// 프로그램시작 

1	int main() {

		// 변수 선언 및 초기화

		char name[32] = "";

		char job[32] = "";

		char hobby[32] = "";

2		char color[32] = "";

	 

		// 친구와 대화하기 

		// 이름 물어보고 답하기 

3		printf( "안녕? 넌 이름이 뭐야: " );

4		scanf( "%s", name );

5		printf( "만나서 반가워, %s. ", name );

	 

		// 직업 물어보고 답하기 

6		printf( "넌 주로 뭐하고 살아: " );

7		scanf( "%s", job );

8		printf( "%s? 나도 마찬가지야 ㅎㅎ ", job );

	 

		// 취미 물어보고 답하기 

9		printf( "그럼 심심할 때 뭐하니: " );

10		scanf( "%s", hobby );

11		printf( "%s? 나랑 똑같네. ", hobby );

	 

		// 좋아하는 색 물어보고 답하기 

12		printf( "혹시 좋아하는 색은 뭐야 : " );

13		scanf( "%s", color );

14		printf( "%s이라구? 와~ 이것도 나랑 똑같네. 신기하다. \n", color );

	 

		// 친하게 지내자고 인사하기 

15		printf( "앞으로 우리 친하게 지내자." );

	 

		// 프로그램 종료 

16		return 0;

	}
