알고리즘

프로그램시작 
   먼저 붉은 태양과 푸른 바다를 출력 
   밤바다로 변경하려면 입력 받기 
   밤바다를 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/Sea.c

	#include <stdio.h>

	// 프로그램시작  

1	int main() {

2		char next = '\0';

	 

		// 먼저 붉은 태양과 푸른 바다를 출력 

3		printf(" \033[31m");

4		printf("   \\_/  \n" );

5		printf("  _/ \\_ \n" );

6		printf("   \\_/  \n" );

7		printf("   / \\  " );

8		printf(" \033[0m \n \n \033[44m");

9		printf("~~~~~~ ~~~~~ ~~~~~~~~ ~~  ~~~~~~~~~~ \n" );

10		printf("~     ~~   ~  ~   ~~~~~~~   ~~~    ~ \n" );

11		printf("  ~ ~~~~~~~      ~~          ~    ~~ \n" );

12		printf("~  ~       ~ ~   ~    ~~~~~~~    ~ ~ \n" );

13		printf("      ~             ~        ~    ~  " );

	 

		// 밤바다로 변경하려면 입력 받기  

14		printf(" \033[0m \n \033[12;1f밤바다를 보려면 문자를 입력하세요: " );

15		next = getchar();

		 

		// 밤바다를 출력 

16		printf(" \033[33m \033[1;3f");

17		printf("   *    *    .      *   .   *    .  \n" );

18		printf("  .    *         .       *.     .   \n" );

19		printf("             .    *         *     * \n" );

20		printf("    *  .       *    .   . *     *   \n" );

21		printf("   .    *        .       .   *    . \033[0m \033[12;1f \n" );

 

		// 프로그램 종료 

22		return 0;

	}
